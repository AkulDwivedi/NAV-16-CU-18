    PROCEDURE ApplySTAmount@1500013();
    VAR
      ServTaxExciseCredit@1500003 : Page 16580;
      ServiceTaxBuffer@1500004 : Record 16476;
      AppliedAmt@1500002 : Decimal;
      AppliedAmteCess@1500001 : Decimal;
      AppliedAmtSHECess@1500000 : Decimal;
    BEGIN
      AppliedAmt := "ST BED Amount" + "ST AED(GSI) Amount" + "ST AED(TTA) Amount" + "ST SED Amount" + "ST SAED Amount" +
        "ST NCCD Amount" + "ST ADET Amount" + "ST ADE Amount";
      AppliedAmteCess := "ST eCESS Amount";
      AppliedAmtSHECess := "ST SHECess Amount";
      IF AppliedAmt + AppliedAmteCess + AppliedAmtSHECess <> 0 THEN BEGIN
        ServTaxExciseCredit.SetDocument("Document No.");
        ServTaxExciseCredit.SetTemplate("Journal Template Name");
        ServTaxExciseCredit.SetBatch("Journal Batch Name");
        ServiceTaxBuffer.RESET;
        ServiceTaxBuffer.SETRANGE("Document No.","Document No.");
        IF ServiceTaxBuffer.FINDSET THEN
          REPEAT
            IF ServiceTaxBuffer."Credit Available" > AppliedAmt THEN BEGIN
              ServiceTaxBuffer."Credit Availed" := AppliedAmt;
              AppliedAmt := 0;
            END;
            IF ServiceTaxBuffer."Credit Available (eCess)" > AppliedAmteCess THEN BEGIN
              ServiceTaxBuffer."Credit Availed (eCess)" := AppliedAmteCess;
              AppliedAmteCess := 0;
            END;
            IF ServiceTaxBuffer."Credit Available (SHECess)" > AppliedAmtSHECess THEN BEGIN
              ServiceTaxBuffer."Credit Availed (SHECess)" := AppliedAmtSHECess;
              AppliedAmtSHECess := 0;
            END;
            IF ServiceTaxBuffer."Credit Available" <= AppliedAmt THEN BEGIN
              ServiceTaxBuffer."Credit Availed" := ServiceTaxBuffer."Credit Available";
              AppliedAmt := AppliedAmt - ServiceTaxBuffer."Credit Available";
            END;
            IF ServiceTaxBuffer."Credit Available (eCess)" <= AppliedAmteCess THEN BEGIN
              ServiceTaxBuffer."Credit Availed (eCess)" := ServiceTaxBuffer."Credit Available (eCess)";
              AppliedAmteCess := AppliedAmteCess - ServiceTaxBuffer."Credit Available (eCess)";
            END;
            IF ServiceTaxBuffer."Credit Available (SHECess)" <= AppliedAmtSHECess THEN BEGIN
              ServiceTaxBuffer."Credit Availed (SHECess)" := ServiceTaxBuffer."Credit Available (SHECess)";
              AppliedAmtSHECess := AppliedAmtSHECess - ServiceTaxBuffer."Credit Available (SHECess)";
            END;
            ServiceTaxBuffer.MODIFY;
          UNTIL ServiceTaxBuffer.NEXT = 0;
      END;
    END;
