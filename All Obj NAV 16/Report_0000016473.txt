    LOCAL PROCEDURE CalcDates@1500004();
    BEGIN
      StartDate := DMY2DATE(1,FromMonth + 1,FromYear);
      EndDate := CALCDATE('<CM>',DMY2DATE(1,ToMonth + 1,ToYear));
    END;

    LOCAL PROCEDURE CheckDataEntry@1280006();
    BEGIN
      CalcDates;
      IF (EndDate < StartDate) OR (EndDate > CALCDATE('<1Y - 1D>',StartDate)) THEN
        ERROR(ErrText003);
    END;

    PROCEDURE GetStartingDateForSTEntry@1500001();
    VAR
      STEntryDetail@1500100 : Record 16474;
      STEntry@1102601000 : Record 16473;
    BEGIN
      StartDate := DMY2DATE(1,FromMonth + 1,FromYear);
      EndDateForInv := EndDate;
      WITH STEntryDetail DO BEGIN
        SETCURRENTKEY("Posting Date");
        SETRANGE(Type,"Purch/Sale");
        SETRANGE("Service Tax Group Code",STGroup);
        SETRANGE("Service Tax Registration No.","ST No.");
        SETRANGE("Posting Date",DMY2DATE(1,FromMonth + 1,FromYear),EndDate);
        IF FINDSET THEN
          REPEAT
            STEntry.GET("Service Tax Entry No.");
            IF STEntry."Posting Date" < StartDate THEN
              StartDate := STEntry."Posting Date";
            IF STEntry."Posting Date" > EndDateForInv THEN
              EndDateForInv := STEntry."Posting Date";
          UNTIL NEXT = 0;
      END;
    END;

    BEGIN
    END.
