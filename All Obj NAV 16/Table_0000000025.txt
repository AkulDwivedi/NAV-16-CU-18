    PROCEDURE CalcAppliedTDSBase@1500002(AppliedAmount@1500001 : Decimal) : Decimal;
    VAR
      TDSEntry@1500000 : Record 13729;
      ApplicationRatio@1500002 : Decimal;
      TDSBaseAmount@1500003 : Decimal;
    BEGIN
      CALCFIELDS(Amount);
      IF Amount = 0 THEN
        EXIT(0);
      ApplicationRatio := ABS(AppliedAmount / Amount);

      TDSEntry.SETRANGE("Transaction No.","Transaction No.");
      TDSEntry.SETRANGE("TDS Nature of Deduction","TDS Nature of Deduction");
      IF TDSEntry.FINDSET THEN
        REPEAT
          IF TDSEntry."TDS Base Amount" = 0 THEN
            TDSBaseAmount += TDSEntry."Work Tax Base Amount"
          ELSE
            TDSBaseAmount += TDSEntry."TDS Base Amount";
        UNTIL TDSEntry.NEXT = 0;
      EXIT(ROUND(TDSBaseAmount * ApplicationRatio));
    END;
