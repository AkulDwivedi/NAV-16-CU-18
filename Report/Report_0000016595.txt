    PROCEDURE FilterJobLedgerEntry@1500000(JobNo@1500001 : Code[20];JobTaskNo@1500002 : Code[20];Type@1500000 : ' ,Labor,Material,Overheads') : Decimal;
    BEGIN
      JobLedgerEntry.RESET;
      JobLedgerEntry.SETCURRENTKEY("Job No.","Job Task No.","Posting Date","CWIP G/L Type","CWIP Posted to G/L");
      JobLedgerEntry.SETRANGE("Job No.",JobNo);
      JobLedgerEntry.SETRANGE("Job Task No.",JobTaskNo);
      JobLedgerEntry.SETFILTER("Posting Date",'<=%1',TillDate);
      JobLedgerEntry.SETRANGE("CWIP G/L Type",Type);
      JobLedgerEntry.SETRANGE("CWIP Posted to G/L",TRUE);
      JobLedgerEntry.CALCSUMS(JobLedgerEntry."Total Cost (LCY)");
      EXIT(JobLedgerEntry."Total Cost (LCY)");
    END;

    PROCEDURE FilterPostedEntry@1500004(JobNo@1500001 : Code[20];JobTaskNo@1500000 : Code[20]);

      PstdCapitalizeCWIPLine@1500002 : Record 13702;
    BEGIN
      PstdCapitalizeCWIPLine.RESET;
      PstdCapitalizeCWIPLine.SETCURRENTKEY("Job No.","Job Task No.","Posting Date");
      PstdCapitalizeCWIPLine.SETRANGE("Job No.",JobNo);
      PstdCapitalizeCWIPLine.SETRANGE("Job Task No.",JobTaskNo);
      PstdCapitalizeCWIPLine.SETFILTER("Posting Date",'<=%1',TillDate);
      PstdCapitalizeCWIPLine.CALCSUMS("Amt to Capitalize (Labor)","Amt to Capitalize (Material)","Amt to Capitalize (Overheads)");
      CWIPAmtPstdLabor := PstdCapitalizeCWIPLine."Amt to Capitalize (Labor)";
      CWIPAmtPstdMaterial := PstdCapitalizeCWIPLine."Amt to Capitalize (Material)";

    END;
