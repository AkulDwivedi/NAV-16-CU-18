    LOCAL PROCEDURE CalculateOpenBalance@1500006(ItemNo@1500000 : Code[20];PostingDate@1170000001 : Date) OpenBalance : Decimal;
    VAR
      ItmLedEntry@1170000000 : Record 32;
    BEGIN
      OpenBalance := 0;
      ItmLedEntry.RESET;
      ItmLedEntry.SETCURRENTKEY("Item No.","Posting Date");
      ItmLedEntry.SETRANGE("Item No.",ItemNo);
      ItmLedEntry.SETRANGE("Posting Date",0D,CALCDATE('<-1d>',PostingDate));
      IF ItmLedEntry.FIND('-') THEN
        REPEAT
          OpenBalance += ItmLedEntry.Quantity;
        UNTIL ItmLedEntry.NEXT = 0;
    END;

    PROCEDURE CheckNewGroup@1500008(VAR ItemLedgerEntry@1500000 : Record 32) : Boolean;
    VAR
      ItemLedgerEntryLocal@1170000000 : Record 32;
      NewGroup@1170000001 : Boolean;
    BEGIN
      NewGroup := FALSE;
      ItemLedgerEntryLocal.COPY(ItemLedgerEntry);
      IF ItemLedgerEntryLocal.NEXT = 1 THEN BEGIN
        IF ItemLedgerEntryLocal."Item No." <> ItemLedgerEntry."Item No." THEN
          NewGroup := TRUE;
      END ELSE
        NewGroup := TRUE;
      EXIT(NewGroup);
    END;

    PROCEDURE InitGroup@1170000003();
    BEGIN

      CloseBal := 0; // Reset Closing Balance when the Item changes.
      InitialOpenBalance := CalculateOpenBalance("Item Ledger Entry"."Item No.","Item Ledger Entry"."Posting Date");
    END;
