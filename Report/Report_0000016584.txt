    PROCEDURE SetInvoiceLineValues@1500001(SalesInvLine3@1500000 : Record 113);
    VAR
      SalesInvLineExtraItem@1500001 : Record 113;
      ExcDutyOnExtraItem@1500002 : Decimal;
    BEGIN
      SalesInvHeader.GET(SalesInvLine3."Document No.");

      SalesInvLineExtraItem.RESET;
      SalesInvLineExtraItem.SETRANGE("Document No.",SalesInvLine3."Document No.");
      SalesInvLineExtraItem.SETRANGE("Re-Dispatch",FALSE);
      IF SalesInvLineExtraItem.FINDSET THEN
        REPEAT
          ExcDutyOnExtraItem += SalesInvLineExtraItem."Excise Amount";
        UNTIL SalesInvLineExtraItem.NEXT = 0;

      IF Counter > 1 THEN
        ExcDutyOnExtraItem := 0;

      IF Cust.GET("Return Receipt Line"."Sell-to Customer No.") THEN BEGIN
        CustomerInfo := Cust.Name + ' ' + Cust.Address + ' ' + Cust."Address 2";
        CustomerEccNo := Cust."E.C.C. No."
      END;

      IF ProcessCarriedOut.GET(SalesInvLine3."Process Carried Out") THEN
        ProcessCarrout := ProcessCarriedOut.Description;

      IF ReturnReason.GET("Return Receipt Line"."Return Reason Code") THEN
        ReturnReasonCode := ReturnReason.Description;

      Identificationmark := "Return Receipt Line"."Identification Mark";
      CenvatCreditAvailed := FORMAT(ROUND("Return Receipt Line"."Excise Amount",0.01,'='));
      InvoiceNo := 'Inv. No: ' + FORMAT(SalesInvLine3."Document No.");
      InvoicePostingdate := 'Date: ' + FORMAT(SalesInvHeader."Posting Date");
      InvoiceQty := 'Qty: ' + FORMAT(SalesInvLine3.Quantity);
      IF Cust.GET(SalesInvLine3."Sell-to Customer No.") THEN
        InvoiceCustomer := 'Cust: ' + Cust.Name;

      IF SalesInvLine3.Quantity = "Return Receipt Line".Quantity THEN
        AdditionalDuty := FORMAT(ROUND(SalesInvLine3."Excise Amount" - "Return Receipt Line"."Excise Amount" +
              ExcDutyOnExtraItem,0.01,'='))
      ELSE
        IF SalesInvLine3.Quantity < "Return Receipt Line".Quantity THEN BEGIN
          ExciseAmountOnSentItems := ("Return Receipt Line"."Excise Amount" / "Return Receipt Line".Quantity) *
            SalesInvLine3.Quantity;
          AdditionalDuty := FORMAT(ROUND(SalesInvLine3."Excise Amount" - ExciseAmountOnSentItems + ExcDutyOnExtraItem,0.01,'='));
        END;

      IF SalesInvLine3."Excise Amount" = "Return Receipt Line"."Excise Amount" THEN BEGIN
        CenvatCrReversed := FORMAT(ROUND(SalesInvLine3."Excise Amount",0.01,'='));
        ExcisableGoodsUsed := 'No';
        IF ExcDutyOnExtraItem > 0 THEN
          ExcisableGoodsUsed := 'Yes'
      END ELSE
        IF SalesInvLine3."Excise Amount" > "Return Receipt Line"."Excise Amount" THEN BEGIN
          CenvatCrReversed := FORMAT(ROUND("Return Receipt Line"."Excise Amount",0.01,'='));
          ExcisableGoodsUsed := 'Yes';
        END
        ELSE
          IF (SalesInvLine3."Excise Amount" < "Return Receipt Line"."Excise Amount") AND
             (SalesInvLine3.Quantity < "Return Receipt Line".Quantity)
          THEN BEGIN
            IF SalesInvLine3."Excise Amount" > ExciseAmountOnSentItems THEN BEGIN
              CenvatCrReversed := FORMAT(ROUND(ExciseAmountOnSentItems,0.01,'='));
              ExcisableGoodsUsed := 'Yes';
            END ELSE
              IF ExcDutyOnExtraItem > 0 THEN BEGIN
                CenvatCrReversed := FORMAT(ROUND(ExciseAmountOnSentItems,0.01,'='));
                ExcisableGoodsUsed := 'Yes';
              END ELSE BEGIN
                CenvatCrReversed := FORMAT(ROUND(SalesInvLine3."Excise Amount",0.01,'='));
                ExcisableGoodsUsed := 'No';
              END;
          END ELSE
            IF (SalesInvLine3."Excise Amount" < "Return Receipt Line"."Excise Amount") AND
               (SalesInvLine3.Quantity = "Return Receipt Line".Quantity)
            THEN BEGIN
              CenvatCrReversed := FORMAT(ROUND(SalesInvLine3."Excise Amount",0.01,'='));
              ExcisableGoodsUsed := 'No';
              IF ExcDutyOnExtraItem > 0 THEN
                ExcisableGoodsUsed := 'Yes';
            END;
    END;
